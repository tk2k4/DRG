    function handleFileUpload(input) {

            const data = new Uint8Array(e.target.result);

            const firstSheetName = workbook.SheetNames[0];
            const worksheet = workbook.Sheets[firstSheetName];

            // Convert to JSON
            const rawData = XLSX.utils.sheet_to_json(worksheet, { raw: false }); // raw: false to read dates/numbers as strings if needed

            // Process and Validate Data

        };

    function parseGender(val) {
        if (!val) return 'N';
        val = String(val).toUpperCase();
        if (val === '1' || val === 'M' || val === 'NAM') return 'M';
        if (val === '2' || val === 'F' || val === 'NU' || val === 'NỮ') return 'F';

    function extractCodes(row, prefix) {
        let codes = [];
        // Check keys starting with prefix (e.g., SDx1, SDx2...)
        Object.keys(row).forEach(key => {
            if (key.startsWith(prefix) && row[key]) {
                codes.push(String(row[key]).trim());
            }
        });
        return codes.join(' ');
    }

    function updateResults(results) {
        results.forEach(res => {
            const rowId = res.id; // internal ID
            const item = excelData.find(x => String(x._id) === rowId);
            if (!item) return;

            // Update UI
            const actualEl = document.getElementById(`actual-${rowId}`);
            const statusEl = document.getElementById(`status-${rowId}`);
            const noteEl = document.getElementById(`note-${rowId}`);

            if (res.error) {
                statusEl.innerHTML = `<span class="px-2 inline-flex text-xs leading-5 font-semibold rounded-full bg-yellow-100 text-yellow-800">Error</span>`;
                noteEl.innerText = res.error;
                item.Status = 'Error';
            } else {
                let actualDRG = res.drg_code;

                // --- Frontend Cleanup: Remove backend HTML error ---
                if (actualDRG && (actualDRG.includes('<span') || actualDRG.includes('Không tìm thấy') || actualDRG.includes('No matching'))) {
                    actualDRG = 'No DRG';
                    // Also mark as Fail if not already
                    item.Status = 'Fail';
                    statusEl.innerHTML = `<span class="px-2 py-0.5 inline-flex text-xs leading-4 font-semibold rounded-full bg-red-500 text-white">No DRG</span>`;
                }

                actualEl.innerText = actualDRG;

                // Compare (simple string contains check or exact match)
                const expected = normalizeDRG(item.Expected_Result);
                const actual = normalizeDRG(actualDRG);

                if (actual.includes(expected) || expected.includes(actual)) {
                    statusEl.innerHTML = `<span class="px-2 inline-flex text-xs leading-5 font-semibold rounded-full bg-green-100 text-green-800">Pass</span>`;
                    item.Status = 'Pass';
                    // Clear System Log if success
                    item.System_Log = '';
                } else {
                    statusEl.innerHTML = `<span class="px-2 inline-flex text-xs leading-5 font-semibold rounded-full bg-red-100 text-red-800">Fail</span>`;
                    item.Status = 'Fail';
                    // Store System Log
                    if (res.explain) {
                        item.System_Log = res.explain;
                        // Show link to view system log
                        noteEl.innerHTML = `<a href="#" onclick="viewLog(\`${escapeHtml(res.explain)}\`); return false;" class="text-blue-500 hover:underline">View Log</a>`;
                    } else if (res.error) {
                        item.System_Log = res.error;
                        noteEl.innerText = res.error;
                    } else {
                        item.System_Log = "Result mismatch";
                    }
                }

                // If saved
                if (res.saved) {
                    noteEl.innerHTML = (noteEl.innerHTML || '') + ` <span class="text-green-600 font-bold ml-1">✓ Saved</span>`;
                }
            }
        });
    }

    function updateSummary() {

            if (item.Status === 'Pass') acc.pass++;
            else if (item.Status === 'Fail') acc.fail++;

            return acc;

        document.getElementById('fail-count').innerText = summary.fail;
        document.getElementById('error-count').innerText = summary.error;

        // Show summary section

    }

    function normalizeDRG(code) {
